template<class CT> void UpdateXandV(CT &x, CT &v, float t, float dt,
	CT(*Acceleration(float t, const CT &x, const CT &v)))
{
	CT a = Acceleration(t, x, v);
	x = x + dt * v;
	v = v + dt * a;
}
glm::vec2 Accel(float t; const glm::vec2 &x, const glm::vec2 &c)
{
	return glm::vec2(-cos(t), -sin(t));
}
